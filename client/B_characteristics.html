接口说明

YJH.BluetoothleManager.characteristics(
	characteristicsSuccess,   \\成功回调 
	characteristicsError,   \\失败回调
	{
		"address": "UUID",   \\特定蓝牙设备的UUID
		"service": "serviceUUID",   \\服务的UUID
		"characteristics": []   \\筛选的特征UUIDs
	}
);
  • 参数
  • 可为空
  • 参数类型
  • 备注
  • address
  • NO
  • String
  • 特定蓝牙设备的UUID
  • service
  • NO
  • String
  • 服务的UUID
  • characteristics
  • YES
  • Array
  • 筛选的特征UUIDs

调用实例:

	
document.addEventListener("deviceready",Deviceready(),false)

function  Deviceready(){
	YJH.BluetoothleManager.characteristics(
	\\成功回调 
	function(succ){   
		var status=succ.status   
		\\获取成功返回的状态，如"characteristics"
		
		var characteristics=succ.characteristics   
		\\获取成功返回的characteristics，值为:
		[
			{ "properties": { "notify":true },"uuid": "2a37" }, 
			{ "properties": { "read": true }, "uuid":"2a38" }
		]
		
		var name=succ.name   
		\\获取成功返回的蓝牙设备名称。如"Polar H7 3B321015"
		
		var services=succ.services   
		\\获取成功返回的services,值为"180d"
		
		var address=succ.address   
		\\获取成功返回的设备地址，如"ECC037FD-72AE-AFC5-9213-CA785B3B5C63"
	},
	
	\\失败回调
	function(err){   
		var error=err.error
		\\返回关闭失败时的状态，可以为"enable"或"initialize"或
		"arguments"或"connect"或"neverConnected"或"discover"。
		
		var message=err.message 
		\\返回关闭失败时的错误信息，可以为"Bluetooth not enabled"或
		"Bluetooth not initialized"或"Argument object not found"或
		"No connection address"或"Never connected to connection"或
		"系统返回的错误信息"。
	},
	
	\\参数
	{
		"address": "UUID",   \\特定蓝牙设备的UUID
		"service": "serviceUUID",   \\服务的UUID
		"characteristics": []   \\筛选的特征UUIDs
	}
  )
}
