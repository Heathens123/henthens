接口说明

在iOS端不能向特征的配置描述写数据.

YJH.BluetoothleManager.writeDescriptor(
	writeDescriptorSuccess,   \\成功回调 
	writeDescriptorError,   \\失败回调 
	{ 
		"address": "addressUUID ",   \\蓝牙UUID 
		"service": "serviceUUID",   \\服务UUID
		"characteristic": "characteristicUUID"   \\特征UUID
		"descriptor": "descriptorUUID"   \\描述UUID
		"value":"WriteValue"   \\写入的值base64编码
		"type":"WriteType"   \\写入的类型
	}
);
  • 参数
  • 可为空
  • 参数类型
  • 备注
  • address
  • NO
  • String
  • 蓝牙UUID
  • service
  • NO
  • String
  • 服务UUID
  • characteristic
  • NO
  • String
  • 特征UUID
  • descriptor
  • NO
  • String
  • 描述UUID
  • value
  • NO
  • String
  • 写入的值base64编码
  • type
  • NO
  • String
  • 写入类型，包括response和noResponse.

调用实例:

			
document.addEventListener("deviceready",Deviceready(),false) 

function  Deviceready(){
	YJH.BluetoothleManager.writeDescriptor(
	\\成功回调
	function(succ){    
		var status=succ.status   
		\\写入成功后返回的status；值为"writeDescriptor"
		
		var service=succ.service   
		\\写入成功后返回的特定服务UUID。如"180d"
		
		var characteristic=succ.characteristic   
		\\写入成功后返回的特征的UUID。如"2a37"
		
		var descriptor=succ.descriptor   
		\\写入成功返回的描述UUID。如"2902"
		
		var value=succ.value   
		\\写入成功返回的写入的值。如"AQAAAAAAAAA="
		
		var address=succ.address   
		\\写入成功返回的蓝牙UUID，如"ABC123"
	},
	
	\\失败回调
	function(err){   
		var error=err.error
		\\返回写入失败时的状态，值为"enable"或"initialize"或"arguments"或
		"connect"或"neverConnected"或"service"或"characteristic"或
		"descriptor"或"readDescriptor"。
		
		var message=err.message 
		\\返回写入失败时的错误信息，值为"Bluetooth not enabled"或
		"Bluetooth not initialized"或"Argument object not found"或
		"No connection address"或"Never connected to connection"或
		"Service not found"或"Characteristic not found"或
		"Descriptor not found"或"系统返回的错误信息"。
		
		var name=err.name   
		\\成功后返回配对的设备名称,此项可以没有
		
		var address=err.address   
		\\成功后返回配对的蓝牙UUID,此项可以没有
	}, 
	
	\\参数
	{
		"service":"180D",   \\服务UUID 
		"characteristic":"2A37",   \\特征UUID
		"descriptor":"2902",   \\描述UUID
		"value":"AQAAAAAAAAA=",   \\写入的值base64编码
		"address":"ABC123"   \\蓝牙UUID 
	}
  );
}
