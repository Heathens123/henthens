接口说明

YJH.H5Dialogs.prompt(
	Message,   \\弹窗的提示信息
	completeCallback,   \\点击按钮后的回调信息
	Title,   \\弹窗的标题
) 
或
YJH.H5Dialogs.prompt(
	Message,   \\弹窗的提示信息
	completeCallback,   \\点击按钮后的回调信息
	Title,   \\弹窗的标题
	buttonLables,   \\按钮数组
	defultText  \\输入框默认信息，默认为空
)
  • 参数
  • 可为空
  • 参数类型
  • 备注
  • Message
  • NO
  • String
  • 提示信息
  • completeCallback
  • NO
  • Function
  • 按钮点击回调
  • Title
  • NO
  • String
  • 标题
  • buttonLabel
  • YES
  • Array
  • 按钮数组，可包含多个,默认为[确认，取消]
  • defultText
  • YES
  • String
  • 输入框默认信息，默认为空

调用实例:

document.addEventListener("deviceready",Deviceready(),false)

function Deviceready(){
	YJH.H5Dialogs.prompt("Message",function(suc){console.log(suc)},"title") 或
	YJH.H5Dialogs.prompt(
		"Message",   \\弹窗的提示信息
		function(suc){   \\点击按钮后的回调信息
			var status=suc.status,   \\点击按钮之后返回的状态
			var des=suc.des,   \\点击按钮之后返回的状态描述
			var result=suc.result   \\点击按钮后返回按钮在按钮数组中的的索引值
			},
		"title",   \\弹窗的标题
		["OK","Cancel"],   \\按钮数组，可包含多个,默认为[确认，取消]
		"aeaa"   \\输入框默认信息，默认为空
	)
}
