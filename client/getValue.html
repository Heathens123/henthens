接口说明.

YJH.H5ModuleManager.getValue(
	successCallBack,   \\成功回调
	FailedCallBack,   \\失败回调
	key   \\key
)
  • 参数
  • 可为空
  • 参数类型
  • 备注
  • successCallBack
  • NO
  • Function
  • 成功回调
  • FailedCallBack
  • NO
  • Function
  • 失败回调
  • key
  • NO
  • String
  • key

调用实例:

document.addEventListener("deviceready",Deviceready(),false)

function Deviceready(){
	YJH.H5ModuleManager.getValue(
	\\成功回调
	function(suc){   
		var status=suc.status,   
		\\获取数据成功的状态，成功返回0
		
		var des=suc.des,   
		\\获取数据成功的状态描述，成功返回"Success"
		
		var result=des.result   
		\\获取成功返回的结果，成功返回{}
	},
	
	\\失败回调
	function(err){   
		var status=err.status,   
		\\获取数据失败的状态，失败返回1
		
		var des=err.des,   
		\\获取数据失败的状态描述，失败返回"can not find value with key:userInao"
		
		var result=err.result   
		\\获取失败返回的结果，失败返回""
	},
	
	"userInfo"   \\key
  )
}
