接口说明.

此方法主要用于设置所有应用都可以访问的数据，通过getValue来获取，注意，不能使用key为appVersion，userInfo，userToken。因为主APP会使用者三个键，避免冲突，系统做了区分。

YJH.H5ModuleManager.setValueForKey(
	successCallBack,   \\成功回调
	FailedCallBack,   \\失败回调
	key,   \\key
	Value   \\需要保存的值
)
  • 参数
  • 可为空
  • 参数类型
  • 备注
  • successCallBack
  • NO
  • Function
  • 成功回调
  • FailedCallBack
  • NO
  • Function
  • 失败回调
  • key
  • NO
  • String
  • key
  • Value
  • NO
  • String,Array或者字典,或者对象
  • 需要保存的值

调用实例:

			
document.addEventListener("deviceready",Deviceready(),false)

function Deviceready(){
	YJH.H5ModuleManager.setValueForKey(
	\\成功回调
	function(suc){
		var status=suc.status,   
		\\设置成功返回的状态，成功返回0
		
		var des=suc.des,   
		\\设置成功返回的状态描述，成功返回"Success"
		
		var result=suc.result   
		\\设置成功返回的结果，成功返回""
	},
	
	\\失败回调
	function(err){
		var status=err.status,   
		\\设置失败返回的状态，失败返回-1
		
		var des=err.des,   
		\\设置失败返回的状态描述，失败返回 "value is cannot be nil"
		
		var result=err.result   
		\\设置失败返回的结果，失败返回""
	},
	
	"key",   \\key
	
	"value"   \\需要保存的值
  )
}
