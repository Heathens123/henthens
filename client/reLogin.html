接口说明

当用户的token信息过期时，可以调用reLogin重新获取token.

YJH.H5NativeAppInfo.reLogin(
	successHander,   \\成功回调
	failedHander   \\失败回调
);
  • 参数
  • 可为空
  • 参数类型
  • 备注
  • successHander
  • NO
  • Function
  • 成功回调
  • failedHander
  • NO
  • Function
  • 失败回调

调用实例:

document.addEventListener("deviceready",Deviceready(),false)

function Deviceready(){
	YJH.H5NativeAppInfo.reLogin(
	\\成功回调
	function(succ){   
		var status=succ.status,   
		\\登录成功返回的状态，成功返回0
		
		var des=succ.des,   
		\\登录成功返回的状态描述，成功返回"success"
		
		var UserToken=succ.result.UserToken   
		\\登录成功返回的UserToken，成功返回一个字符串比如:
		"d0c391a116cf57c69072b0a015d55368"
		
		var UserInfo=succ.result.UserInfo   
		\\登陆成功返回用户的个人信息，如用户的账户信息等:
		{account: "18583702682", isReal: 0, areaName: "", 
		clientType: 0, appType: 2,等}
	},
	
	\\失败回调
	function(err){   
		var status=succ.status,   
		\\登录失败返回的状态，失败返回1
		
		var des=succ.des,   
		\\登录失败返回的状态描述，失败返回"failed"
		
		var result=succ.result   
		\\登录失败返回的结果，失败返回 "用户取消登录"
	}
  );
}
