接口说明 

YJH.BluetoothleManager.retrieveConnected(
	retrieveConnectedSuccess,   \\成功回调
	retrieveConnectedError,   \\失败回调 
	{ 
		"services"：serviceId  \\需要重新配对的设备serviceId 
	}
);
  • 参数
  • 可为空
  • 参数类型
  • 备注
  • services
  • YES
  • Array
  • 需要重新配对的设备serviceId.如果传递的是空数组，则会重新连接所有配对过的设备.

调用实例:

document.addEventListener("deviceready",Deviceready(),false)

function Deviceready(){
	YJH.BluetoothleManager.retrieveConnected(
	\\成功回调
	function(succ){   
		var name=succ[0].name,   
		\\成功后返回配对成功的设备名称
		var address=succ[0].address   
		\\成功后返回配对成功的蓝牙地址
	},
	
	\\失败回调
	function(err){   
		var error=err.error,  
	    \\返回失败状态，可能的值为"enable"或者"initialize"
	    
		var message=err.message
	    \\返回错误信息，可能的值为"Bluetooth not enabled"或者
	     "Bluetooth not initialized"
	},
	
	\\参数
	{ 
		"services": [ "180D", "180F" ]  \\需要重新配对的设备serviceId  
	}
  );
}
