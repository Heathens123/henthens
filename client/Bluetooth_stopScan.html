接口说明

停止扫描蓝牙设备，因为扫描设备相对来说比较耗电，所以当发现我们想要的设备后，请调用此方法，停止扫描。

YJH.BluetoothleManager.stopScan(
	stopScanSuccess,   \\成功回调
	stopScanError   \\失败回调
);
  • 参数
  • 可为空
  • 参数类型
  • 备注
  • 无
  • 无
  • 无
  • 无

调用实例:

document.addEventListener("deviceready",Deviceready(),false)

function Deviceready(){
	YJH.BluetoothleManager.stopScan(
	
		\\成功回调
		function(succ){   
			var status=succ.status   \\成功时返回的状态
		}, 
		
		\\失败回调
		function(err){   
			var error=err.error   
			\\失败时返回的状态，可能的值为"enable"或者"initialize"或者"startScan";
			
			var message=err.message  
			\\失败时返回的错误信息，可能的值为"Bluetooth not enabled"或者
			"Bluetooth not initialized"或者"Not scanning"
		}
	);
}
