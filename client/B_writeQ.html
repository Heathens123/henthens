接口说明

此方法主要用于写入超过20字节的数据.

YJH.BluetoothleManager.write(
	writeSuccess,   \\成功回调 
	writeError,   \\失败回调 
	{ 
		"value":"WriteValue"   \\写入的值
		"service": "serviceUUID",   \\服务UUID 
		"characteristic": "characteristicUUID"   \\特征UUID
		"type":"WriteType"   \\写入的类型
		"address": "addressUUID ",   \\蓝牙UUID 
	}
);
  • 参数
  • 可为空
  • 参数类型
  • 备注
  • address
  • NO
  • String
  • 蓝牙UUID
  • service
  • NO
  • String
  • 服务UUID
  • characteristic
  • NO
  • String
  • 特征UUID
  • value
  • NO
  • String
  • 写入的值
  • type
  • NO
  • String
  • 写入类型，包括response和noResponse.

调用实例:

document.addEventListener("deviceready",Deviceready(),false) 

function  Deviceready(){
	YJH.BluetoothleManager.write(
    \\成功回调
	function(succ){    
		var status=succ.status   
		\\获取成功后返回的status；值为"written"
		
		var service=succ.service   
		\\获取成功后返回的特定服务UUID。如"180d"
		
		var characteristic=succ.characteristic   
		\\获取成功后返回的特征的UUID。如"2a37"
		
		var value=succ.value   
		\\获取成功返回的写入的值。如"V3JpdGUgSGVsbG8gV29ybGQ="
		
		var address=succ.address   
		\\获取成功返回的蓝牙UUID，如"ECC037FD-72AE-AFC5-9213-CA785B3B5C63"
	},
	
	\\失败回调
	function(err){   
		var error=err.error
		\\返回关闭失败时的状态，值为"enable"或"initialize"或"arguments"或
		"connect"或"neverConnected"或"service"或"characteristic"
		或"descriptors"或"subscription"或write""。
		
		var message=err.message 
		\\返回关闭失败时的错误信息，值为"Bluetooth not enabled"或
		"Bluetooth not initialized"或"Argument object not found"或
		"No connection address"或"Never connected to connection"或
		"Service not found"或"Characteristic not found"或"系统返回的错误信息"
		或"Already subscribed"或"Write value not found"或"系统返回错误信息"。
		
		var name=err.name   
		\\成功后返回配对的设备名称,此项可以没有
		
		var address=err.address   
		\\成功后返回配对的蓝牙UUID,此项可以没有
	},
	
	\\参数
	{
		"value":"V3JpdGUgSGVsbG8gV29ybGQ=",  \\写入的值
		"service":"180F",   \\服务UUID 
		"characteristic":"2A19",   \\特征UUID
		"type":"noResponse",   \\写入的类型
		"address":"ABC123"   \\蓝牙UUID 
	}
  );
}
